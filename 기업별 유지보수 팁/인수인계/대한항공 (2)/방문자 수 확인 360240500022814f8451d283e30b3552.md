### 총 방문자수 조회

SELECT COUNT(*) AS cnt_2025
-> FROM tbl_websight_log
-> WHERE wdate >= '2025-01-01 00:00:00'
->   AND wdate <  '2026-01-01 00:00:00';

![image.png](%EB%B0%A9%EB%AC%B8%EC%9E%90%20%EC%88%98%20%ED%99%95%EC%9D%B8/image%203.png)

### 일일 최대 방문자 수 및 해당 일자

SELECT day, hits
-> FROM (
->   SELECT DATE(wdate) AS day, COUNT(*) AS hits
->   FROM tbl_websight_log
->   WHERE wdate >= '2025-01-01 00:00:00'
->     AND wdate <  '2026-01-01 00:00:00'
->   GROUP BY DATE(wdate)
-> ) d
-> ORDER BY hits DESC
-> LIMIT 1;

![image.png](%EB%B0%A9%EB%AC%B8%EC%9E%90%20%EC%88%98%20%ED%99%95%EC%9D%B8/image%204.png)

### 일별 방문자수

SELECT DATE(wdate) AS day, COUNT(*) AS hits
-> FROM tbl_websight_log
-> WHERE wdate >= '2025-01-01 00:00:00'
->   AND wdate <  '2026-01-01 00:00:00'
-> GROUP BY DATE(wdate)
-> ORDER BY day;
