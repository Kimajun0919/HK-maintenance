### 일일 최대 방문자 수 및 해당 일자

SELECT day, hits
-> FROM (
->   SELECT DATE(wdate) AS day, COUNT(*) AS hits
->   FROM tbl_websight_log
->   WHERE wdate >= '2025-01-01 00:00:00'
->     AND wdate <  '2026-01-01 00:00:00'
->   GROUP BY DATE(wdate)
-> ) d
-> ORDER BY hits DESC
-> LIMIT 1;
